核心代码解析

HTML 转 Markdown

def html_to_markdown(self, html_content, note_title):
    """将 HTML 转换为 Markdown"""
    # 使用 BeautifulSoup 预处理
    soup = BeautifulSoup(html_content, 'lxml')
    
    # 处理代码块
    for pre in soup.find_all('pre'):
        code = pre.find('code')
        if code:
            language = code.get('class', [''])[0].replace('language-', '')
            code.string = f"\n```{language}\n{code.get_text()}\n```\n"
    
    # 转换为 Markdown
    markdown = self.h2t.handle(str(soup))
    
    return markdown.strip()

图片路径转换

def convert_image_paths(self, markdown, note_name, assets_subfolder):
    """转换图片路径为 Obsidian 格式"""
    # 匹配 Markdown 图片语法 ![alt](path)
    pattern = r'!\[([^\]]*)\]\(([^)]+)\)'
    
    def replace_path(match):
        alt = match.group(1)
        original_path = match.group(2)
        
        # 提取文件名
        filename = os.path.basename(original_path)
        
        # 构建新路径：assets/note_name/filename
        new_path = f"assets/{assets_subfolder}/{filename}"
        
        return f'![{alt}]({new_path})'
    
    return re.sub(pattern, replace_path, markdown)

YAML 头部生成

def create_yaml_header(self, note_info):
    """创建 YAML Front Matter"""
    # 处理标签
    tags = []
    if note_info['tags']:
        tags = [tag.strip() for tag in note_info['tags'].split(',')]
    
    # 构建 YAML
    yaml_lines = [
        '---',
        f'title: "{note_info["title"]}"',
        f'date: {note_info["created"]}',
        f'modified: {note_info["modified"]}',
    ]
    
    if tags:
        yaml_lines.append(f'tags: [{", ".join(tags)}]')
    
    yaml_lines.append('source: WizNote')
    yaml_lines.append('---\n')
    
    return '\n'.join(yaml_lines)

核心技术：Turndown.js

Turndown 配置示例

const TurndownService = require('turndown');

const turndownService = new TurndownService({
    headingStyle: 'atx',
    codeBlockStyle: 'fenced',
    bulletListMarker: '-'
});

// 添加代码块语言识别
turndownService.addRule('codeBlock', {
    filter: 'pre',
    replacement: function(content, node) {
        const code = node.querySelector('code');
        let language = '';
        
        if (code) {
            const classes = code.className.split(' ');
            language = classes.find(c => c.startsWith('language-'))
                ?.replace('language-', '') || '';
        }
        
        return '\n```' + language + '\n' + content + '\n```\n';
    }
});

// 手动添加语言标记
turndownService.addRule('code', {
    filter: function(node) {
        return node.nodeName === 'CODE' && 
               node.parentNode.nodeName === 'PRE';
    },
    replacement: function(content, node) {
        // 从 class 中提取语言
        const className = node.className || '';
        const langMatch = className.match(/language-(\w+)/);
        const lang = langMatch ? langMatch[1] : '';

    }

def convert_mathjax_to_latex(html_content):
    """转换 MathJax 为 LaTeX"""
    # 行内公式
    html_content = re.sub(
        r'<script type="math/tex">(.*?)</script>',
        r'$\1$',
        html_content,
        flags=re.DOTALL
    )
    
    # 块级公式
    html_content = re.sub(
        r'<script type="math/tex; mode=display">(.*?)</script>',
        r'$$\1$$',
        html_content,
        flags=re.DOTALL
    )
    
    return html_content

问题 6：标签丢失

原因：标签未转换为 YAML

解决方案：

def extract_tags_from_db(db_path, note_guid):
    """从数据库提取标签"""
    conn = sqlite3.connect(db_path)
    cursor = conn.execute(
        "SELECT DOCUMENT_TAGS FROM WIZ_DOCUMENT WHERE DOCUMENT_GUID = ?",
        (note_guid,)
    )
    result = cursor.fetchone()
    conn.close()
    
    if result and result[0]:
        return [tag.strip() for tag in result[0].split(',')]
    return []

def add_tags_to_yaml(markdown, tags):
    """添加标签到 YAML 头部"""
    if tags:
        yaml_tags = f'tags: [{", ".join(tags)}]\n'
        # 在 YAML 头部插入 tags
        markdown = re.sub(
            r'(---\n.*?)(---)',
            r'\1' + yaml_tags + r'\2',
            markdown,
            count=1,
            flags=re.DOTALL
        )
    return markdown

问题 7：笔记创建时间错误

原因：数据库时间戳格式与 YAML 不一致

解决方案：

from datetime import datetime

def convert_wiz_time(wiz_time_str):
    """转换为 Obsidian 时间格式"""
    # 为知笔记时间格式：2024-01-15 10:30:45
    # Obsidian 格式：2024-01-15 或 2024-01-15T10:30:45
    
    try:
        dt = datetime.strptime(wiz_time_str, '%Y-%m-%d %H:%M:%S')
        return dt.strftime('%Y-%m-%d')
    except:
        return datetime.now().strftime('%Y-%m-%d')

问题 8：特殊字符转义

原因：Markdown 特殊字符未转义

解决方案：

def escape_markdown_chars(text):
    """转义 Markdown 特殊字符"""
    # 不在代码块中的特殊字符
    special_chars = ['\\', '`', '*', '_', '{', '}', '[', ']', '(', ')', '#', '+', '-', '.', '!', '|']
    
    for char in special_chars:
        text = text.replace(char, '\\' + char)
    
    return text

Python 依赖安装

# 安装核心依赖
pip install beautifulsoup4 lxml html2text

# 可选依赖（图片处理）
pip install pillow

# 验证安装
python -c "import bs4, lxml, html2text; print('✓ 依赖安装成功')"

本项目文件清单

./
├── 为知笔记到Obsidian迁移方案.md    # 完整迁移方案文档
├── wiznote-to-obsidian-migration.html # 本网页版本
├── wiz2obsidian.py                  # Python 迁移脚本
├── validate_migration.py            # 验证脚本
├── run_migration.bat                # Windows 快速启动脚本
├── README.md                        # 工具包说明
└── README_迁移工具.md               # 详细使用指南

✅ 迁移前检查清单

  • 已备份为知笔记原始数据
  • 已安装 Python 3.9+ 或 Node.js 18+
  • 已安装必需依赖库
  • 为知笔记数据已同步到本地
  • 已创建 Obsidian 仓库目录
  • 已检查加密笔记（需要先解密）
